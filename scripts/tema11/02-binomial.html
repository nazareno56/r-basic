from scipy.stats import binom 
import matplotlib.pyplot as plt
import numpy as np

fig, ax = plt.subplots(1,1)
n = 7
p = 0.4

mean, var, skew, kurt = binom.stats(n, p, moments = 'mvsk')

print("Media %f"%mean)
## Media 2.800000

## Varianza 1.680000

## Sesgo 0.154303

## Curtosis -0.261905

ax.legend(loc = 'best', frameon = False)
